country_event = { 
	id = rcpmap.2
	fire_only_once = no
	hidden = yes
	is_triggered_only = yes
	immediate = {
		name = rcpmap.2.a
		
		every_owned_state = {
			limit = {
				has_state_flag = rcpmap_owned
			} 
			every_neighbor_state = {
				limit = {
					NOT = {
						has_state_flag = rcpmap_owned
					} 
					NOT = {
						is_core_of = ROOT
					}
					NOT = {
						has_state_category = wasteland
					}
				} 
				set_state_flag = rcpmap_owned 
				ROOT = {transfer_state = PREV} 
				add_core_of = PREV
			}
		}
		random_owned_state = {	
			limit = {
				has_state_flag = rcpmap_owned
			} 
			ROOT = {
				set_capital = PREV
			}
		}
	}

}
